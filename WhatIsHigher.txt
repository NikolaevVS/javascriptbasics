/* Это моё первое приложение на JavaScript, которое считает числа
*/
let a = prompt('Введите число a') // Присваиваем значение a
let b = prompt('Введите число b') // Присваиваем значение b
if (a>b) {
	console.log('Больше число а'); // Условие
}
else if (a == b) { 
	console.log('Числа равны');
}
else {
  console.log('Больше число b');
}
